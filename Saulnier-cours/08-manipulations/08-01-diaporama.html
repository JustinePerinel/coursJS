2 - Un 2nd diaporama...

Ce second diaporama est fait avec une fonction anonyme dans un écouteur d'évènement, dans les scripts en bas de page

document.getElementById('btnDiapo2').addEventListener('click', function() {
document.getElementById('imgDiapo2').src = 'img/ile-perdue.jpg';
document.getElementById('imgDiapo2').alt = 'la vue est belle';
})

- Un diaporama avec une fonction nommée

document.getElementById('btnDiapo3').addEventListener('click', imgDiapo3);
function imgDiapo3() {
document.getElementById('imgDiapo3').src = 'img/tortue.jpg';
document.getElementById('imgDiapo3').alt = 'la tortue';
}

3 - Un 3ème diaporama

avec un switch dans une fonction...

function tourneImages() {
var getImage = document.getElementById('imgDiapo4').getAttribute('src')
console.log(getImage);

switch(getImage) {
case 'img/falaise.jpg':
document.getElementById('imgDiapo4').src = 'img/plage1.jpg'
break;
}
switch(getImage) {
case 'img/plage1.jpg':
document.getElementById('imgDiapo4').src = 'img/pluie1.jpg'
break;
}
switch(getImage) {
case 'img/pluie1.jpg':
document.getElementById('imgDiapo4').src = 'img/falaise.jpg'
break;
}
}
